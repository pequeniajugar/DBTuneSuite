table1:
n = input size
orders(ordernum, itemnum,  quantity, storeid, vendorid);
	ordernum: int	1...n		primary key
	itemnum: int	(1, n / 2.5)
	quantity: int	(1, 100)
	storeid:  text	 (1, n // 100)
	vendorid: text	 (1...n+1)

CREATE TABLE orders (
    ordernum INT,
    itemnum INT,          
    quantity INT, 
    storeid VARCHAR(20),   
    vendorid VARCHAR(20)       
);


table2:
n = input size // 100
store( storeid, name );
	storeid: text	1...n		primary key
	name: text	Store1...Storen 

CREATE TABLE store (
    storeid VARCHAR(20),
    name VARCHAR(25)
);

n = input size / 2.5

	price: decimal.2	(1, 100) 

CREATE TABLE item (
